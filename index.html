  <script>

    // Set up the canvas for confetti

    const canvas = document.getElementById('confettiCanvas');

    const ctx = canvas.getContext('2d');

    let W = window.innerWidth;

    let H = window.innerHeight;

    canvas.width = W;

    canvas.height = H;


    window.addEventListener('resize', () => {

      W = window.innerWidth;

      H = window.innerHeight;

      canvas.width = W;

      canvas.height = H;

    });


    // Create confetti particles

    const confettiCount = 150;

    const confetti = [];


    for (let i = 0; i < confettiCount; i++) {

      confetti.push({

        x: Math.random() * W,

        y: Math.random() * H - H,

        r: Math.random() * 6 + 4,

        d: Math.random() * confettiCount,

        color: "hsl(" + Math.floor(Math.random() * 360) + ", 100%, 50%)",

        tilt: Math.floor(Math.random() * 10) - 10,

        tiltAngleIncremental: (Math.random() * 0.07) + 0.05,

        tiltAngle: 0

      });

    }


    // Draw confetti particles

    function drawConfetti() {

      ctx.clearRect(0, 0, W, H);

      confetti.forEach(p => {

        ctx.beginPath();

        ctx.lineWidth = p.r;

        ctx.strokeStyle = p.color;

        ctx.moveTo(p.x + p.tilt + (p.r / 2), p.y);

        ctx.lineTo(p.x + p.tilt, p.y + p.tilt + (p.r / 2));

        ctx.stroke();

      });

      updateConfetti();

    }


    // Update confetti positions

    function updateConfetti() {

      confetti.forEach(p => {

        p.tiltAngle += p.tiltAngleIncremental;

        p.y += (Math.cos(p.d) + 3 + p.r / 2) / 2;

        p.x += Math.sin(p.d);

        p.tilt = Math.sin(p.tiltAngle) * 15;


        // Reset particle when it goes off-screen

        if (p.y > H) {

          p.y = -10;

          p.x = Math.random() * W;

        }

      });

    }


    let confettiInterval;

    function startConfetti() {

      confettiInterval = setInterval(drawConfetti, 20);

      // Stop the confetti after 5 seconds

      setTimeout(stopConfetti, 5000);

    }

    function stopConfetti() {

      clearInterval(confettiInterval);

      ctx.clearRect(0, 0, W, H);

    }


    // Start celebration: trigger confetti and play music

    function startCelebration() {

      startConfetti();

      playMusic();

    }


    // Play a birthday tune (a sample royalty-free track)

    function playMusic() {

      const audio = new Audio('https://www.bensound.com/bensound-music/bensound-hey.mp3');

      audio.play();

    }
